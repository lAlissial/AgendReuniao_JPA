CREATE TABLE REUNIAO (ID  SERIAL NOT NULL, ASSUNTO VARCHAR(255), DATAHORA TIMESTAMP, PRIMARY KEY (ID))
CREATE TABLE PARTICIPANTE (NOME VARCHAR(255) NOT NULL, DTYPE VARCHAR(31), EMAIL VARCHAR(255), EMPRESA VARCHAR(255), PRIMARY KEY (NOME))
CREATE TABLE REUNIAO_PARTICIPANTE (reunioes_ID INTEGER NOT NULL, participantes_NOME VARCHAR(255) NOT NULL, PRIMARY KEY (reunioes_ID, participantes_NOME))
ALTER TABLE REUNIAO_PARTICIPANTE ADD CONSTRAINT FK_REUNIAO_PARTICIPANTE_reunioes_ID FOREIGN KEY (reunioes_ID) REFERENCES REUNIAO (ID)
ALTER TABLE REUNIAO_PARTICIPANTE ADD CONSTRAINT FK_REUNIAO_PARTICIPANTE_participantes_NOME FOREIGN KEY (participantes_NOME) REFERENCES PARTICIPANTE (NOME)
